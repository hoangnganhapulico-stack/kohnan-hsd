<script>

    let currentMode = 'date';


    function setMode(mode) {

        currentMode = mode;

        document.getElementById('opt-date').classList.toggle('active', mode === 'date');

        document.getElementById('opt-duration').classList.toggle('active', mode === 'duration');

        document.getElementById('input-exp-date').classList.toggle('hidden', mode !== 'date');

        document.getElementById('input-exp-duration').classList.toggle('hidden', mode !== 'duration');

    }


    function checkHSD() {

        const mfgInput = document.getElementById('date-mfg').value;

        const itemType = document.getElementById('item-type').value;

        const resBox = document.getElementById('result-box');

        const resStatus = document.getElementById('result-status');

        const resDetail = document.getElementById('result-detail');

        const resDateLabel = document.getElementById('result-date');

        const today = new Date();

        today.setHours(0,0,0,0);


        if (!mfgInput) {

            alert("Vui lòng nhập Ngày Sản Xuất!");

            return;

        }


        const mfgDate = new Date(mfgInput);

        let expDate;


        if (currentMode === 'date') {

            const expInput = document.getElementById('date-exp').value;

            if (!expInput) { alert("Vui lòng nhập Ngày Hết Hạn!"); return; }

            expDate = new Date(expInput);

        } else {

            const val = parseInt(document.getElementById('duration-val').value);

            const unit = document.getElementById('duration-unit').value;

            if (isNaN(val)) { alert("Vui lòng nhập số lượng thời gian (Tháng/Năm)!"); return; }

            

            expDate = new Date(mfgDate);

            if (unit === 'months') {

                expDate.setMonth(expDate.getMonth() + val);

            } else {

                expDate.setFullYear(expDate.getFullYear() + val);

            }

        }


        // Tính toán

        const totalLife = expDate - mfgDate;

        const remainingLife = expDate - today;

        const percentage = (remainingLife / totalLife) * 100;

        const limit = itemType === 'vn' ? 65 : 50;


        resBox.style.display = 'block';

        resDateLabel.innerText = "Hết hạn ngày: " + expDate.toLocaleDateString('vi-VN');


        if (remainingLife < 0) {

            resBox.className = "expired";

            resStatus.innerText = "❌ HÀNG HẾT HẠN";

            resDetail.innerText = "Tuyệt đối không nhận hàng đã quá hạn sử dụng.";

        } else if (percentage >= limit) {

            resBox.className = "pass";

            resStatus.innerText = "✅ NHẬN HÀNG";

            resDetail.innerHTML = `Tỷ lệ đạt <b>${percentage.toFixed(1)}%</b><br>(Yêu cầu tối thiểu > ${limit}%)`;

        } else {

            resBox.className = "fail";

            resStatus.innerText = "🚫 TRẢ HÀNG";

            resDetail.innerHTML = `Tỷ lệ chỉ còn <b>${percentage.toFixed(1)}%</b><br>(Thấp hơn mức quy định ${limit}%)`;

        }

    }

</script>
