    <script>

        function calculateHSD() {

            const mfgDateVal = document.getElementById('mfgDate').value;

            const expDateVal = document.getElementById('expDate').value;

            const type = document.getElementById('type').value;

            const resDiv = document.getElementById('result');

            const resStatus = document.getElementById('res-status');

            const resDetail = document.getElementById('res-detail');


            if (!mfgDateVal || !expDateVal) {

                alert("Vui lòng nhập đầy đủ ngày NSX và HSD!");

                return;

            }


            const mfg = new Date(mfgDateVal);

            const exp = new Date(expDateVal);

            const today = new Date();

            today.setHours(0, 0, 0, 0);


            if (exp <= mfg) {

                alert("Ngày hết hạn phải sau ngày sản xuất!");

                return;

            }


            const totalLife = exp - mfg;

            const remainingLife = exp - today;

            const percentage = (remainingLife / totalLife) * 100;

            const required = (type === 'vn') ? 65 : 50;


            resDiv.style.display = 'block';


            if (remainingLife <= 0) {

                resDiv.className = "status-expired";

                resStatus.innerText = "🚨 QUÁ HẠN";

                resDetail.innerText = "Hàng đã hết hạn sử dụng. Không được nhận.";

            } else if (percentage >= required) {

                resDiv.className = "status-pass";

                resStatus.innerText = "✅ ĐẠT (NHẬN)";

                resDetail.innerHTML = `Tỉ lệ HSD còn lại: <b>${percentage.toFixed(1)}%</b><br>(Yêu cầu tối thiểu: ${required}%)`;

            } else {

                resDiv.className = "status-fail";

                resStatus.innerText = "❌ KHÔNG ĐẠT (TRẢ)";

                resDetail.innerHTML = `Tỉ lệ HSD còn lại: <b>${percentage.toFixed(1)}%</b><br>(Thấp hơn yêu cầu: ${required}%)`;

            }

        }

    </script>
